## 2️⃣ **Dicionario.java** - Recurso Compartilhado Principal

```java
package modelo;

import util.Config;
import java.io.BufferedReader;
import java.io.FileReader;
import java.io.IOException;
import java.util.HashMap;
import java.util.Map;
import java.util.concurrent.Semaphore;
import java.util.concurrent.atomic.AtomicInteger;

public class Dicionario {
    private Map<String, String> palavras;
    private Semaphore semaforo;
    private AtomicInteger totalAcessos;
    private AtomicInteger totalEsperas;
    
    public Dicionario(int numPermissoes) {
        this.palavras = new HashMap<>();
        this.semaforo = new Semaphore(numPermissoes, true); // true = justo (evita starvation)
        this.totalAcessos = new AtomicInteger(0);
        this.totalEsperas = new AtomicInteger(0);
        carregarDicionario();
    }
    
    private void carregarDicionario() {
        try (BufferedReader br = new BufferedReader(
                new FileReader(Config.CAMINHO_DICIONARIO))) {
            
            String linha;
            int contador = 0;
            
            while ((linha = br.readLine()) != null) {
                linha = linha.trim();
                if (linha.isEmpty() || linha.startsWith("#")) {
                    continue; // Ignora linhas vazias e comentários
                }
                
                String[] partes = linha.split(";");
                if (partes.length == 2) {
                    String palavraPt = partes[0].trim().toLowerCase();
                    String palavraEn = partes[1].trim().toLowerCase();
                    palavras.put(palavraPt, palavraEn);
                    contador++;
                }
            }
            
            System.out.println("[DICIONÁRIO] Carregado com " + contador + " palavras");
            
        } catch (IOException e) {
            System.err.println("[ERRO] Não foi possível carregar o dicionário: " + e.getMessage());
            System.err.println("[INFO] Criando dicionário básico...");
            carregarDicionarioBasico();
        }
    }
    
    private void carregarDicionarioBasico() {
        // Dicionário mínimo caso o arquivo não exista
        palavras.put("o", "the");
        palavras.put("a", "the");
        palavras.put("um", "a");
        palavras.put("uma", "a");
        palavras.put("e", "and");
        palavras.put("de", "of");
        palavras.put("em", "in");
        palavras.put("para", "to");
        palavras.put("que", "that");
        palavras.put("com", "with");
        palavras.put("cachorro", "dog");
        palavras.put("gato", "cat");
        palavras.put("casa", "house");
        palavras.put("livro", "book");
        palavras.put("era", "was");
        palavras.put("vez", "time");
        palavras.put("tinha", "had");
        palavras.put("morava", "lived");
        palavras.put("grande", "big");
        palavras.put("feliz", "happy");
        
        System.out.println("[DICIONÁRIO] Dicionário básico criado com " + palavras.size() + " palavras");
    }
    
    public String traduzir(String palavra, int threadId) throws InterruptedException {
        boolean precisouEsperar = false;
        
        // Tenta adquirir permissão
        if (!semaforo.tryAcquire()) {
            precisouEsperar = true;
            totalEsperas.incrementAndGet();
            
            if (Config.EXIBIR_PROGRESSO_DETALHADO) {
                System.out.println("  [Thread " + threadId + "] ⏳ Aguardando acesso ao dicionário...");
            }
            
            semaforo.acquire(); // Bloqueia até conseguir
        }
        
        try {
            totalAcessos.incrementAndGet();
            
            if (Config.EXIBIR_PROGRESSO_DETALHADO && precisouEsperar) {
                System.out.println("  [Thread " + threadId + "] ✓ Conseguiu acesso ao dicionário");
            }
            
            // Simula tempo de busca no dicionário
            Thread.sleep(Config.DELAY_TRADUCAO_MS);
            
            // Traduz ou mantém original entre colchetes
            String palavraLimpa = limparPalavra(palavra);
            String traducao = palavras.getOrDefault(palavraLimpa.toLowerCase(), "[" + palavraLimpa + "]");
            
            return traducao;
            
        } finally {
            semaforo.release();
        }
    }
    
    private String limparPalavra(String palavra) {
        // Remove pontuação mas mantém a palavra
        return palavra.replaceAll("[.,;:!?\"']", "");
    }
    
    public int getTotalAcessos() {
        return totalAcessos.get();
    }
    
    public int getTotalEsperas() {
        return totalEsperas.get();
    }
    
    public int getTamanhoDicionario() {
        return palavras.size();
    }
}
```

## 4️⃣ **EstatisticasTraducao.java** - Métricas do processo

```java
package modelo;

import java.util.concurrent.atomic.AtomicInteger;
import java.util.concurrent.atomic.AtomicLong;

public class EstatisticasTraducao {
    private AtomicInteger palavrasTraduzidas;
    private AtomicLong tempoTotalMs;
    private int totalPalavras;
    private long tempoInicio;
    
    public EstatisticasTraducao(int totalPalavras) {
        this.palavrasTraduzidas = new AtomicInteger(0);
        this.tempoTotalMs = new AtomicLong(0);
        this.totalPalavras = totalPalavras;
        this.tempoInicio = System.currentTimeMillis();
    }
    
    public void incrementarPalavrasTraduzidas() {
        palavrasTraduzidas.incrementAndGet();
    }
    
    public void finalizarContagem() {
        long tempoFim = System.currentTimeMillis();
        tempoTotalMs.set(tempoFim - tempoInicio);
    }
    
    public int getPalavrasTraduzidas() {
        return palavrasTraduzidas.get();
    }
    
    public int getTotalPalavras() {
        return totalPalavras;
    }
    
    public long getTempoTotalMs() {
        return tempoTotalMs.get();
    }
    
    public double getTempoTotalSegundos() {
        return tempoTotalMs.get() / 1000.0;
    }
    
    public int getPercentualCompleto() {
        if (totalPalavras == 0) return 0;
        return (palavrasTraduzidas.get() * 100) / totalPalavras;
    }
    
    public double getPalavrasPorSegundo() {
        long tempo = System.currentTimeMillis() - tempoInicio;
        if (tempo == 0) return 0;
        return (palavrasTraduzidas.get() * 1000.0) / tempo;
    }
}
```

## 5️⃣ **TarefaTraducao.java** - Thread que traduz um bloco

```java
package tarefa;

import modelo.BlocoTexto;
import modelo.Dicionario;
import modelo.EstatisticasTraducao;
import util.Config;

public class TarefaTraducao implements Runnable {
    private int id;
    private BlocoTexto bloco;
    private Dicionario dicionario;
    private EstatisticasTraducao estatisticas;
    private StringBuilder resultado;
    private int palavrasTraduzidas;
    private boolean concluida;
    
    public TarefaTraducao(int id, BlocoTexto bloco, Dicionario dicionario, 
                          EstatisticasTraducao estatisticas) {
        this.id = id;
        this.bloco = bloco;
        this.dicionario = dicionario;
        this.estatisticas = estatisticas;
        this.resultado = new StringBuilder();
        this.palavrasTraduzidas = 0;
        this.concluida = false;
    }
    
    @Override
    public void run() {
        try {
            System.out.println("[Thread " + id + "] 🚀 Iniciou - " + bloco);
            
            for (String palavra : bloco.getPalavras()) {
                // Traduz a palavra (aqui ocorre a concorrência!)
                String traducao = dicionario.traduzir(palavra, id);
                resultado.append(traducao).append(" ");
                
                // Atualiza estatísticas
                palavrasTraduzidas++;
                estatisticas.incrementarPalavrasTraduzidas();
            }
            
            concluida = true;
            System.out.println("[Thread " + id + "] ✅ Concluída - " + 
                             palavrasTraduzidas + " palavras traduzidas");
            
        } catch (InterruptedException e) {
            System.out.println("[Thread " + id + "] ⚠️ Interrompida!");
            Thread.currentThread().interrupt();
        } catch (Exception e) {
            System.err.println("[Thread " + id + "] ❌ Erro: " + e.getMessage());
        }
    }
    
    public String getResultado() {
        return resultado.toString().trim();
    }
    
    public int getProgresso() {
        int total = bloco.getTamanho();
        if (total == 0) return 100;
        return (palavrasTraduzidas * 100) / total;
    }
    
    public boolean estaConcluida() {
        return concluida;
    }
    
    public int getId() {
        return id;
    }
    
    public BlocoTexto getBloco() {
        return bloco;
    }
}
```

## 6️⃣ **MonitorProgresso.java** - Exibe progresso no terminal

```java
package controle;

import modelo.EstatisticasTraducao;
import tarefa.TarefaTraducao;
import util.Config;

import java.util.List;

public class MonitorProgresso implements Runnable {
    private List<TarefaTraducao> tarefas;
    private EstatisticasTraducao estatisticas;
    private volatile boolean ativo;
    
    public MonitorProgresso(List<TarefaTraducao> tarefas, EstatisticasTraducao estatisticas) {
        this.tarefas = tarefas;
        this.estatisticas = estatisticas;
        this.ativo = true;
    }
    
    @Override
    public void run() {
        try {
            while (ativo && !todasConcluidas()) {
                limparTela();
                exibirProgresso();
                Thread.sleep(Config.INTERVALO_ATUALIZACAO_MS);
            }
            
            // Exibição final
            limparTela();
            exibirProgresso();
            
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
        }
    }
    
    private void exibirProgresso() {
        System.out.println("╔════════════════════════════════════════════════════════════════╗");
        System.out.println("║          TRADUTOR CONCORRENTE DE LIVROS - PROGRESSO            ║");
        System.out.println("╚════════════════════════════════════════════════════════════════╝");
        System.out.println();
        
        // Progresso geral
        int percentualGeral = estatisticas.getPercentualCompleto();
        String barraGeral = gerarBarra(percentualGeral, 50);
        
        System.out.println("📊 PROGRESSO GERAL: " + percentualGeral + "%");
        System.out.println(barraGeral);
        System.out.println("   Palavras: " + estatisticas.getPalavrasTraduzidas() + 
                         "/" + estatisticas.getTotalPalavras());
        System.out.println("   Velocidade: " + 
                         String.format("%.1f", estatisticas.getPalavrasPorSegundo()) + 
                         " palavras/segundo");
        System.out.println();
        
        // Progresso por thread
        System.out.println("🧵 THREADS:");
        for (TarefaTraducao tarefa : tarefas) {
            int progresso = tarefa.getProgresso();
            String barra = gerarBarra(progresso, 30);
            String status = tarefa.estaConcluida() ? "✅ CONCLUÍDA" : "⚙️  PROCESSANDO";
            
            System.out.println("   [Thread " + tarefa.getId() + "] " + status);
            System.out.println("   " + barra + " " + progresso + "%");
        }
        
        System.out.println();
        System.out.println("─────────────────────────────────────────────────────────────────");
    }
    
    private String gerarBarra(int percentual, int tamanho) {
        int completo = (percentual * tamanho) / 100;
        int incompleto = tamanho - completo;
        
        StringBuilder barra = new StringBuilder("[");
        for (int i = 0; i < completo; i++) {
            barra.append("█");
        }
        for (int i = 0; i < incompleto; i++) {
            barra.append("░");
        }
        barra.append("]");
        
        return barra.toString();
    }
    
    private boolean todasConcluidas() {
        return tarefas.stream().allMatch(TarefaTraducao::estaConcluida);
    }
    
    private void limparTela() {
        // Limpa a tela (funciona na maioria dos terminais)
        System.out.print("\033[H\033[2J");
        System.out.flush();
    }
    
    public void parar() {
        ativo = false;
    }
}
```

```java
package util;

import java.io.*;
import java.nio.charset.StandardCharsets;

public class LeitorArquivo {
    
    public static String lerTextoCompleto(String caminhoArquivo) throws IOException {
        StringBuilder conteudo = new StringBuilder();
        
        try (BufferedReader br = new BufferedReader(
                new InputStreamReader(
                    new FileInputStream(caminhoArquivo), 
                    StandardCharsets.UTF_8))) {
            
            String linha;
            while ((linha = br.readLine()) != null) {
                conteudo.append(linha).append(" ");
            }
        }
        
        return conteudo.toString().trim();
    }
    
    public static void escreverTexto(String caminhoArquivo, String texto) throws IOException {
        // Cria diretórios se não existirem
        File arquivo = new File(caminhoArquivo);
        arquivo.getParentFile().mkdirs();
        
        try (BufferedWriter bw = new BufferedWriter(
                new OutputStreamWriter(
                    new FileOutputStream(caminhoArquivo), 
                    StandardCharsets.UTF_8))) {
            
            bw.write(texto);
        }
    }
    
    public static int contarPalavras(String texto) {
        if (texto == null || texto.trim().isEmpty()) {
            return 0;
        }
        return texto.trim().split("\\s+").length;
    }
}
```

## 8️⃣ **GerenciadorTraducao.java** - Orquestra todo o processo

```java
package controle;

import modelo.BlocoTexto;
import modelo.Dicionario;
import modelo.EstatisticasTraducao;
import tarefa.TarefaTraducao;
import util.Config;

import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;

public class GerenciadorTraducao {
    private Dicionario dicionario;
    private int numThreads;
    
    public GerenciadorTraducao() {
        this.numThreads = Config.NUM_THREADS;
        this.dicionario = new Dicionario(Config.PERMISSOES_DICIONARIO);
    }
    
    public String traduzir(String textoCompleto) throws InterruptedException {
        // 1. Prepara estatísticas
        String[] palavras = textoCompleto.split("\\s+");
        EstatisticasTraducao estatisticas = new EstatisticasTraducao(palavras.length);
        
        System.out.println("\n╔════════════════════════════════════════════════════════════════╗");
        System.out.println("║              INICIANDO TRADUÇÃO CONCORRENTE                    ║");
        System.out.println("╚════════════════════════════════════════════════════════════════╝");
        System.out.println("📖 Total de palavras: " + palavras.length);
        System.out.println("🧵 Número de threads: " + numThreads);
        System.out.println("🔒 Acessos simultâneos ao dicionário: " + Config.PERMISSOES_DICIONARIO);
        System.out.println("📚 Tamanho do dicionário: " + dicionario.getTamanhoDicionario() + " palavras");
        System.out.println();
        
        // 2. Divide em blocos
        List<BlocoTexto> blocos = dividirEmBlocos(palavras);
        
        // 3. Cria as tarefas
        List<TarefaTraducao> tarefas = new ArrayList<>();
        for (int i = 0; i < numThreads && i < blocos.size(); i++) {
            tarefas.add(new TarefaTraducao(i + 1, blocos.get(i), dicionario, estatisticas));
        }
        
        // 4. Inicia o monitor de progresso
        MonitorProgresso monitor = new MonitorProgresso(tarefas, estatisticas);
        Thread threadMonitor = new Thread(monitor);
        threadMonitor.start();
        
        // 5. Inicia as threads de tradução
        List<Thread> threads = new ArrayList<>();
        for (TarefaTraducao tarefa : tarefas) {
            Thread thread = new Thread(tarefa);
            threads.add(thread);
            thread.start();
        }
        
        // 6. Aguarda todas terminarem
        for (Thread thread : threads) {
            thread.join();
        }
        
        // 7. Para o monitor
        monitor.parar();
        threadMonitor.join();
        
        // 8. Finaliza estatísticas
        estatisticas.finalizarContagem();
        
        // 9. Junta os resultados
        String resultado = juntarResultados(tarefas);
        
        // 10. Exibe estatísticas finais
        exibirEstatisticasFinais(estatisticas);
        
        return resultado;
    }
    
    private List<BlocoTexto> dividirEmBlocos(String[] palavras) {
        List<BlocoTexto> blocos = new ArrayList<>();
        int palavrasPorBloco = (int) Math.ceil((double) palavras.length / numThreads);
        
        for (int i = 0; i < numThreads; i++) {
            int inicio = i * palavrasPorBloco;
            int fim = Math.min(inicio + palavrasPorBloco, palavras.length);
            
            if (inicio < palavras.length) {
                List<String> palavrasBloco = Arrays.asList(
                    Arrays.copyOfRange(palavras, inicio, fim)
                );
                blocos.add(new BlocoTexto(i, palavrasBloco, inicio));
            }
        }
        
        return blocos;
    }
    
    private String juntarResultados(List<TarefaTraducao> tarefas) {
        StringBuilder textoCompleto = new StringBuilder();
        
        for (TarefaTraducao tarefa : tarefas) {
            textoCompleto.append(tarefa.getResultado()).append(" ");
        }
        
        return textoCompleto.toString().trim();
    }
    
    private void exibirEstatisticasFinais(EstatisticasTraducao estatisticas) {
        System.out.println("\n╔════════════════════════════════════════════════════════════════╗");
        System.out.println("║              TRADUÇÃO CONCLUÍDA COM SUCESSO                    ║");
        System.out.println("╚════════════════════════════════════════════════════════════════╝");
        System.out.println("✅ Palavras traduzidas: " + estatisticas.getPalavrasTraduzidas());
        System.out.println("⏱️  Tempo total: " + 
                         String.format("%.2f", estatisticas.getTempoTotalSegundos()) + " segundos");
        System.out.println("⚡ Velocidade média: " + 
                         String.format("%.1f", estatisticas.getPalavrasPorSegundo()) + 
                         " palavras/segundo");
        System.out.println("🔍 Total de acessos ao dicionário: " + dicionario.getTotalAcessos());
        System.out.println("⏳ Vezes que threads precisaram esperar: " + dicionario.getTotalEsperas());
        System.out.println();
    }
}
```
